<?php
namespace TieuMinh\SumUp1\Block\Pagging;
use Magento\Framework\View\Element\Template;
use Magento\Framework\View\Element\Template\Context;
use TieuMinh\SumUp1\Model\Post;
use Magento\Framework\Pricing\Helper\Data as priceHelper;
class Collection extends Template
{
    protected $customCollection;
    protected $priceHepler;
    public function __construct(Context $context, Post $customCollection,priceHelper $priceHepler)
    {

        $this->priceHepler = $priceHepler;

}
